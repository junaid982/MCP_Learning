>>> python -m venv env-mcp-server

For Windows System  
>>> env-mcp-server\Scripts\activate

For Linux
>>> source env-mcp-server/bin/activate



# 2 - Install requirements.txt

>>> pip install -r requirements.txt

>>> python server.py

>>> python client.py
